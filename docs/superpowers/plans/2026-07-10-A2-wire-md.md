```python
def build_md_context(chunks: list[dict]) -> str:
    blocks = []
    for i, c in enumerate(chunks, start=1):
        title = c.get("title") or ""
        source = c.get("source") or title
        content = c.get("excerpt") or c.get("content") or ""
        blocks.append(f"[P{i}] Tai lieu MD - {title} - doan {c.get('chunk_index')}\nNguon MD: {source}\n{content}")
    return "\n\n".join(blocks)
```

Trong `retrieve_md_knowledge`, sau khi dựng `chunks` (mỗi chunk hiện có title/content/source/chunk_index/page), thêm `excerpt` = content và trả thêm `context`. Sửa đoạn tạo chunks + return:

```python
    chunks = [{"title": r["title"], "content": r["content"], "excerpt": r["content"],
               "source": r["source"], "chunk_index": i, "page": None} for i, r in enumerate(rows)]
    ...
    return {"context": build_md_context(chunks), "chunks": chunks, "quotes": quotes,
            "selected_count": len(chunks), "candidate_count": len(chunks), "top_score": top}
```

import unittest
from cogs.ai import QuoteExtractor, Planner


class QuoteFactTest(unittest.TestCase):
    def test_fact_author_is_trusted(self):
        meta = {"chunks": [], "quotes": [
            {"quote": "Nghệ thuật không cần là ánh trăng lừa dối.", "author": "Nam Cao", "source": "s", "title": "t"}

        plan = Planner.build("nhận định của Nam Cao")
        ev = QuoteExtractor.extract(meta, plan, "nhận định của Nam Cao")
        self.assertEqual(ev[0].author, "Nam Cao")
        self.assertEqual(ev[0].confidence, 1.0)

    def test_author_filter_excludes_other_fact(self):
        meta = {"chunks": [], "quotes": [
            {"quote": "Câu A.", "author": "Nam Cao", "source": "s", "title": "t"},
            {"quote": "Câu B.", "author": "Nguyễn Minh Châu", "source": "s", "title": "t"},
        ]}
        plan = Planner.build("nhận định của Nam Cao")
        ev = QuoteExtractor.extract(meta, plan, "nhận định của Nam Cao")
        authors = {e.author for e in ev}
        self.assertEqual(authors, {"Nam Cao"})
